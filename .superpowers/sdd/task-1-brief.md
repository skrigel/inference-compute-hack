### Step 3: Create slugify utility

Create `frontend/src/lib/slugify.ts`:

```typescript
export function slugify(name: string): string {
  return name
    .toLowerCase()
    .replace(/[^a-z0-9]+/g, "-")
    .replace(/^-|-$/g, "")
    .slice(0, 50);
}

export function generateId(): string {
  return crypto.randomUUID();
}
```
